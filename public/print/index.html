// Initial state
const initialState = {
  error: null,
  loading: false,
  profile: null,
};
​
export default reducer = (state = initialState, action) => {
  const { type, payload } = action;
​
  switch (type) {
    case "FETCH_PROFILE_REQUEST":
      return {
        ...state,
        error: null,
        loading: true,
        profile: null,
      };
    case "FETCH_PROFILE_SUCCESS":
      return {
        ...state,
        error: false,
        loading: false,
        profile: payload.profile,
      };
    case "FETCH_PROFILE_FAILURE":
      return {
        ...state,
        error: payload.error,
        loading: false,
        profile: null,
      };
    default:
      return state;
  }
};

<CodeSurfer>

```reason title="Parser"
// Will return a tree
let make = tokens => {
  let rec parse = tokens => {
    switch (Array.length(tokens)) {
    | 0 => [||]
    | _ =>
      let (parsedDef, remainingTokens) = parseDef(tokens);
      Array.concat([[|parsedDef|], parse(remainingTokens)]);
    };
    // Start with parseDef
    let (parsedDef, _) = parseDef(tokens);
    [|parsedDef|];
  };
  parse(tokens);
};
```
